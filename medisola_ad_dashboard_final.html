데이터 최종 업데이트일 계산


function getLastDataDate(){
  if(!RD.daily || !RD.daily.length) return null;

  const validDates = RD.daily
    .map(r => r.date || r.dt || r.day)
    .filter(Boolean)
    .map(d => new Date(d));

  const lastDate = new Date(Math.max(...validDates));

  const y = lastDate.getFullYear();
  const m = String(lastDate.getMonth()+1).padStart(2,'0');
  const d = String(lastDate.getDate()).padStart(2,'0');

  return `${y}.${m}.${d}`;
}
